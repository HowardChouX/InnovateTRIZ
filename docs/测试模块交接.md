import pytest
from src.core.triz_engine import LocalTRIZEngine, TRIZEngine, get_triz_engine

class TestLocalTRIZEngine:
    """本地TRIZ引擎测试"""

    def test_detect_parameters(self):
        """测试参数检测"""
        engine = LocalTRIZEngine()
        result = engine.detect_parameters("手机需要更大电池但要保持轻薄")
        assert "improving" in result
        assert "worsening" in result
        assert result["improving"] != result["worsening"]

    def test_generate_solutions(self):
        """测试解决方案生成"""
        engine = LocalTRIZEngine()
        solutions = engine.generate_solutions(
            principle_ids=[1, 10, 15],
            problem="测试问题",
            count=3
        )
        assert len(solutions) == 3
        assert all(sol.confidence > 0 for sol in solutions)

    def test_categorize_solutions(self):
        """测试解决方案分类"""
        engine = LocalTRIZEngine()
        solutions = engine.generate_solutions([1, 2, 3, 35, 36], "测试", count=5)
        categorized = engine.categorize_solutions(solutions)
        assert "物理" in categorized or "化学" in categorized

    def test_empty_problem(self):
        """测试空问题处理"""
        engine = LocalTRIZEngine()
        solutions = engine.generate_solutions([1], "", count=1)
        assert len(solutions) > 0  # 应有默认输出

        assert client.client is not None

    def test_is_available(self):
        """测试可用性检查"""
        client = AIClient(api_key=None)
        assert client.is_available() == False

    @pytest.mark.asyncio
    async def test_detect_parameters(self):
        """测试参数检测"""
        client = AIClient(api_key="test-key")

        # Mock API响应
        with patch.object(client, 'client') as mock_client:
            mock_response = MagicMock()
            mock_response.choices = [
                MagicMock(message=MagicMock(content='{"improving": "速度", "worsening": "能耗"}'))

            result = await client.detect_parameters("手机电池问题")
            assert result["improving"] == "速度"
            assert result["worsening"] == "能耗"

    @pytest.mark.asyncio
    async def test_generate_solutions(self):
        """测试解决方案生成"""
        from src.data.models import AIAnalysisRequest

        client = AIClient(api_key="test-key")
        request = AIAnalysisRequest(
            problem="测试问题",
            principle_ids=[1, 10],
            solution_count=2
        )

        with patch.object(client, 'client') as mock_client:
            mock_response = MagicMock()

                MagicMock(message=MagicMock(content='[{"principle_id": 1, "principle_name": "分割原理"}]'))

    """本地存储测试"""

    def temp_storage(self):
